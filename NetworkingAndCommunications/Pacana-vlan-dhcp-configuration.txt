this is my network topology... 

* multilayer switch0:
- (Gig0/2) is connected to multilayer switch1(Gig0/2), 
- (Gig0/1) is connected to switch0(Gig0/1), 
- (Fa0/24) is connected to switch1(Fa0/24).
 
* multilayer switch1:
- (Gig0/2) is connected to multilayer switch0(Gig0/2),
- (Fa0/24) is connected to switch0(Fa0/24),
- (Gig0/1) is connected to switch1(Gig0/1).


* switch0:
- (Gig0/1) is connected to multilayer switch0(Gig0/1), 
- (Fa0/24) is connected to multilayer switch1(Fa0/24),
- (Gig0/2) is connected to switch1(Gig0/2),
- (Fa0/23) is connected to access point0,
- (Fa0/1) is connected to pc0,
- (Fa0/2) is connected to pc1.

* switch1:
- (Fa0/24) is connected to multilayer switch0(Fa0/24),
- (Gig0/1) multilayer switch1(Gig0/1),
- (Gig0/2) is connected to switch0(Gig0/2),
- (Fa0/1) is connected to pc2,
- (Fa0/2) is connected to pc3,
- (Fa0/3) is connected to pc4.
 
* access point:
 is connected to switch0(Fa0/23), and wirelessly to laptop0.


(on multilayer switch 0[CORE — routing + DHCP]):----------------------------------------------------------
enable
configure terminal

! --- VLANs + names ---
vlan 10
 name Employees
exit
vlan 20
 name Admins
exit
vlan 30
 name Wireless
exit

! --- SVIs (gateway IPs for each VLAN) ---
interface Vlan10
 ip address 192.168.10.1 255.255.255.0
 no shutdown
exit

interface Vlan20
 ip address 192.168.20.1 255.255.255.0
 no shutdown
exit

interface Vlan30
 ip address 192.168.30.1 255.255.255.0
 no shutdown
exit

! Enable L3 routing
ip routing

! --- DHCP pools on core ---
ip dhcp excluded-address 192.168.10.1 192.168.10.10
ip dhcp excluded-address 192.168.20.1 192.168.20.10
ip dhcp excluded-address 192.168.30.1 192.168.30.10

ip dhcp pool VLAN10
 network 192.168.10.0 255.255.255.0
 default-router 192.168.10.1
 dns-server 8.8.8.8
exit

ip dhcp pool VLAN20
 network 192.168.20.0 255.255.255.0
 default-router 192.168.20.1
 dns-server 8.8.8.8
exit

ip dhcp pool VLAN30
 network 192.168.30.0 255.255.255.0
 default-router 192.168.30.1
 dns-server 8.8.8.8
exit

! --- Trunk links (adjust interface names if your hardware differs) ---
! Gi0/2 -> multilayer switch1 Gi0/2 (trunk)
interface GigabitEthernet0/2
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 switchport trunk allowed vlan add 20
 switchport trunk allowed vlan add 30
 no shutdown
exit

! Gi0/1 -> switch0 Gi0/1 (trunk)
interface GigabitEthernet0/1
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 switchport trunk allowed vlan add 20
 switchport trunk allowed vlan add 30
 no shutdown
exit

! Fa0/24 -> switch1 Fa0/24 (trunk)
interface FastEthernet0/24
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10
 switchport trunk allowed vlan add 20
 switchport trunk allowed vlan add 30
 no shutdown
exit
