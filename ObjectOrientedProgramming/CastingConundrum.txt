
public class ShapeCalculator {
	
	public static void main(String[] args) {
        Shape[] shapes = {
            new Circle(5),
            new Square(4),
            new Triangle(3, 4)
        };

        calculateAndPrint(shapes);
    }
	
    static void calculateAndPrint(Shape[] shapes) {
        for (Shape shape : shapes) {
            if (shape instanceof Circle) {
                double area = ((Circle) shape).getArea();
                System.out.println("Circle - Area: " + area);
            } else if (shape instanceof Square) {
                double perimeter = ((Square) shape).getSide();
                System.out.println("Square - Perimeter: " + perimeter);
            } else if (shape instanceof Triangle) {
                double hypotenuse = ((Triangle) shape).getlength();
                System.out.println("Triangle - Hypotenuse Length: " + hypotenuse);
            }
        }
    }
    
}

interface Shape{
	void display();
}


public class Circle implements Shape{
	private double radius;
	
	public Circle() {
		
	}
	public Circle(double radius) {
		this.radius = radius;
	}
	
	public double getArea() {
		return 3.14*radius*radius;
	}
	public void display() {
		System.out.printf("Circle - Area: " + getArea());
	}
}

public class Square implements Shape{
	private double side;
	
	public Square() {
		
	}
	
	public Square(double side) {
		this.side = side;
	}
	
	public double getSide() {
		return 4*side;
	}
	
	public void display() {
		System.out.printf("Square - Perimeter: " + getSide());
	}

	
}

public class Triangle implements Shape{
	private double side1;
	private double side2;
	
	public Triangle() {
		
	}
	
	public Triangle(double side1, double side2) {
		this.side1 = side1;
		this.side2 = side2;
	}
	
	public double getlength() {
		return Math.sqrt( side1 * side1 + side2 *side2);
	}
	
	public void display() {
		System.out.printf("Triangle - Hypotenuse Length: " + getlength());
	}
}
