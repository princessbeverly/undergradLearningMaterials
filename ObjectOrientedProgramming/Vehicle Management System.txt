	private int year;
	
	public Vehicle() {

	public Vehicle(String model, int year) {
		this.model = model;
		this.year = year;
	}
	
	//Accessor Methods
	public String getmodel() {
		return model;
	}
	
	public int getyear() {
		return year;
	}
	
}
